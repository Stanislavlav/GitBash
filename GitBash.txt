pwd -- показать где мы на данный момент
mkdir -- создать папку
cd name -- зайти в указанную папку
mkdir name name_1 name_2 -- создать три папки с именами name name_1 name_2
touch name -- создать файл без перехода в него
ls -la - выводит список содержимого директории
cat name - просмотр файла
vim name -- текстовый редактор
nano name -- текстовый редактор попроще
cat >> name -- записать в файл в терминале
cd .. -- выйти на папку выше
mv name namefolder -- переместить файл в папку
cp -- копирование файла
head name -- показать первые строки файла
tail name -- показать последние строки файла
less name -- просмотр содержимоо дилнного файла
